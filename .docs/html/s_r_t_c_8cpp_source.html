63void sRTC::init(sRTC_time_format_t format, sRTC_time_t* time, sRTC_mode_t mode)
64{
65 // Enable RTC clock
66 RCC->CSR |= RCC_CSR_RTCEN;
67
68 // Disable RTC backup register write protection
69 disableBackupWP();
70
71 // Disable write protection
72 disableWP();
73
74 // Enable init phase
75 enableInit();
76
77 // Set RTC A-prescaler to 127 and S-prescaler to 255(default values)
78 handle->PRER = 0x007F00FF;
79
80 // Set time format and disable shadow bypass
81 handle->CR = 0x00000000;
82 handle->CR |= (format << RTC_CR_FMT_Pos) | (0 << RTC_CR_BYPSHAD_Pos);
83
84 // If time pointer is not NULL
85 if (time)
86 {
87 // Set RTC time
88 setDate(time->day, time->month, time->year, time->weekDay);
89 setTime(time->hour, time->minute, time->second, time->ampm);
90 }
91
92 // Enable write protection
93 enableWP();
94
95 // Disable init phase
96 disableInit();
97
98 // Enable RTC backup register write protection
99 enableBackupWP();
100
101 // Enable RTC EXTI line
102 EXTI->IMR |= EXTI_IMR_IM20;
103 EXTI->RTSR |= EXTI_IMR_IM20;
104
105 // Enable RTC IRQ
106 NVIC_SetPriority(RTC_IRQn, 0);
107 NVIC_EnableIRQ(RTC_IRQn);
108
109 // Set halt mode when in DEBUG mode
110 if (mode == sRTC_mode_t::HALT) DBGMCU->APB1FZ |= DBGMCU_APB1_FZ_DBG_RTC_STOP;
111}

156void sRTC::enableWakeup(sRTC_WUT_clock_t clock, uint16_t reloadValue)
157{
158 // Disable RTC backup register write protection
159 disableBackupWP();
160
161 // Disable write protection
162 disableWP();
163
164 // Disable wake up timer
165 handle->CR &= ~(RTC_CR_WUTE);
166
167 // Wait for write flag to set
168 while (!(handle->ISR & RTC_ISR_WUTWF));
169
170 // Clear wake up write interrupt flag
171 handle->ISR &= ~RTC_ISR_WUTWF;
172
173 // Set wake up reload value
174 handle->WUTR = reloadValue - 1;
175
176 // Clear wake up timer clock prescaler
177 handle->CR &= ~RTC_CR_WUCKSEL;
178
179 // Set wake up timer clock prescaler
180 handle->CR |= clock;
181
182 // Clear interrupt flag
183 handle->ISR &= ~RTC_ISR_WUTF;
184
185 // Enable wake up timer interrupt
186 handle->CR |= RTC_CR_WUTIE;
187
188 // Enable wake up from RTC
189 PWR->CSR |= PWR_CSR_WUF;
190
191 // Enable wake up timer
192 handle->CR |= RTC_CR_WUTE;
193
194 // Enable RTC write protection
195 enableWP();
196
197 // Enable RTC backup register write protection
198 enableBackupWP();
199}

201void sRTC::disableWakeup(void)
202{
203 // Disable RTC backup register write protection
204 disableBackupWP();
205
206 // Disable write protection
207 disableWP();
208
209 // Disable wake up timer and its interrupt
210 handle->CR &= ~(RTC_CR_WUTE | RTC_CR_WUTIE);
211
212 // Reset wake up timer counter
213 handle->WUTR = 0;
214
215 // Enable RTC write protection
216 enableWP();
217
218 // Enable RTC backup register write protection
219 enableBackupWP();
220}

224void sRTC::setDate(uint8_t day, uint8_t month, uint8_t year, sRTC_day_t weekDay)
225{
226 // Reset date register to default value
227 handle->DR = 0x00002101;
228
229 // Convert input values to BCD
230 day = dec2BCD(day);
231 month = dec2BCD(month);
232 year = dec2BCD(year);
233
234 // Set RTC date
235 handle->DR = (weekDay << RTC_DR_WDU_Pos) | (day << RTC_DR_DU_Pos) | (month << RTC_DR_MU_Pos) | (year << RTC_DR_YU_Pos);
236}

238void sRTC::setTime(uint8_t hour, uint8_t minute, uint8_t second, sRTC_ampm_t ampm)
239{
240 // Reset time register to default value
241 handle->TR = 0x00000000;
242
243 // Convert input values to BCD
244 hour = dec2BCD(hour);
245 minute = dec2BCD(minute);
246 second = dec2BCD(second);
247
248 // Set RTC time
249 handle->TR = (ampm << RTC_TR_PM_Pos) | (hour << RTC_TR_HU_Pos) | (minute << RTC_TR_MNU_Pos) | (second << RTC_TR_SU_Pos);
250}

284static uint8_t dec2BCD(uint8_t value)
285{
286 // Return decimal value converted to BCD
287 return (((value / 10) << 4) + (value % 10));
288}
289
290static uint8_t BCD2Dec(uint8_t value)
291{
292 // Return BCD value converted to decimal
293 return (((value >> 4) * 10) + (value & 0x0F));
294}
